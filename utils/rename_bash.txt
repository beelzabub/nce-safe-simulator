#!/usr/bin/env bash
# Append .txt to every non-.txt file in the repo, or reverse that operation.
# Usage: bash rename_bash.txt [rename|reverse]
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(dirname "$SCRIPT_DIR")"
MANIFEST="$SCRIPT_DIR/original_files.txt"

usage() {
    echo "Usage: bash $(basename "${BASH_SOURCE[0]}") [rename|reverse]"
    echo "  rename  Snapshot file list, then append .txt to every non-.txt repo file"
    echo "  reverse Restore every repo file to its original name"
}

cmd_rename() {
    if [[ -f "$MANIFEST" ]]; then
        echo "Error: manifest already exists — run 'reverse' first." >&2
        exit 1
    fi

    # Capture only non-.txt files before creating the manifest.
    mapfile -t files < <(find "$REPO_ROOT" -type f \
        -not -path "*/.git/*" \
        -not -path "*/.claude/*" \
        -not -name "*.txt" | sort)

    printf '%s\n' "${files[@]}" > "$MANIFEST"
    echo "Manifest written: $MANIFEST  (${#files[@]} files)"

    for f in "${files[@]}"; do
        [[ -f "$f" ]] && mv "$f" "${f}.txt"
    done

    echo "Done."
}

cmd_reverse() {
    if [[ ! -f "$MANIFEST" ]]; then
        echo "Error: manifest not found at $MANIFEST" >&2
        exit 1
    fi

    count=0
    while IFS= read -r original; do
        [[ -z "$original" ]] && continue
        if [[ -f "${original}.txt" ]]; then
            mv "${original}.txt" "$original"
            count=$((count + 1))
        fi
    done < "$MANIFEST"

    rm "$MANIFEST"
    echo "Done. Restored $count files. Manifest removed."
}

case "${1:-rename}" in
    rename)  cmd_rename ;;
    reverse) cmd_reverse ;;
    *)       usage; exit 1 ;;
esac
